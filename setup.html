OHCA 品管系統 Ver.6 試算表設定

此頁用來建立新版 Google Sheet 工作頁，欄位順序與 WebApp 匯出的 CSV / Excel 相同。完成部署後，WebApp 的「送出」會把資料寫入 OHCA Ver6 工作頁。

設定步驟

  1. 開啟目前使用的 Google 試算表。
  2. 選擇「擴充功能」→「Apps Script」。
  3. 將下方程式碼貼到 Apps Script，儲存後先執行 setupSheet。
  4. 部署為網頁應用程式，權限選「任何人」。
  5. 將部署網址貼回 WebApp 的 components/PreviewModal.tsx 內 GOOGLE_SCRIPT_URL。

function setupSheet() {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = spreadsheet.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = spreadsheet.insertSheet(SHEET_NAME);
  }

  sheet.clear();
  sheet.getRange(1, 1, 1, EXPORT_HEADERS.length).setValues([EXPORT_HEADERS]);
  sheet.setFrozenRows(1);
  sheet.autoResizeColumns(1, EXPORT_HEADERS.length);
}

function doPost(e) {
  const payload = JSON.parse(e.postData.contents || '{}');
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = spreadsheet.getSheetByName(SHEET_NAME);
  if (!sheet) {
    setupSheet();
    sheet = spreadsheet.getSheetByName(SHEET_NAME);
  }

  const record = payload.orderedRecord || {};
  const row = EXPORT_HEADERS.map(header => record[header] ?? '');
  sheet.appendRow(row);

  return ContentService
    .createTextOutput(JSON.stringify({ ok: true, sheet: SHEET_NAME }))
    .setMimeType(ContentService.MimeType.JSON);
}
